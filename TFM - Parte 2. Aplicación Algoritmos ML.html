import sklearn.metrics as metrics


def ROC_AUC(y_test, Model_Probs, Model):
    
    ns_probs = [0 for _ in range(len(y_test))]
    model_probs = Model_Probs[:, 1]
    
    ns_auc = metrics.roc_auc_score(y_test, ns_probs)
    model_auc = metrics.roc_auc_score(y_test, model_probs)
    
    # Mostramos el Area debajo de la curva:
    print('Default: ROC AUC=%.3f' % (ns_auc))
    print(Model+': ROC AUC=%.3f' % (model_auc))
    # Se c√°lcula la curva Roc
    ns_fpr, ns_tpr, _ = metrics.roc_curve(y_test, ns_probs, pos_label= 'Y')
    model_fpr, model_tpr, _ = metrics.roc_curve(y_test, model_probs, pos_label= 'Y')
    # Se a√±aden las curvas al gr√°fico
    plt.plot(ns_fpr, ns_tpr, linestyle='--', label='Default')
    plt.plot(model_fpr, model_tpr, marker='.', label=Model)
    # A√±ade las etiquetas a los ejes
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    # Muestra la leyenda
    plt.legend()
    # Muestra el gr√°fico
    plt.show()
